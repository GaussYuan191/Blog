# 对象扁平化

主要是通过递归，使嵌套的数据扁平

```js
// 创建判断对象的函数
function isObject(obj) {
    return Object.prototype.toString.call(obj).slice(8, -1) === 'Object'
}
function flatObject (obj, pre = '', res = {}) {
    if (!obj) return;
    Reflect.ownKeys(obj).forEach((key) => {
        let perKey = pre + key;
        if (isObject(obj[key])) {
            flatObject(obj[key], perKey + '.', res);
        } else {
            res[perKey] = obj[key];
        }
    })  
    return res;
}
```
